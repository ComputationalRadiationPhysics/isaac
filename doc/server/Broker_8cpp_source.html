137 void Broker::receiveVideo(InsituConnectorGroup* group,uint8_t* video_buffer,char* payload)
138 {
139  //Search for : in payload
140  char* colon = strchr(payload, ':');
141  if (colon == NULL)
142  return;
143  colon++;
144  //Search for ; in payload
145  char* semicolon = strchr(colon, ';');
146  if (semicolon == NULL)
147  return;
148  //Search for , in payload
149  char* comma = strchr(semicolon, ',');
150  if (comma == NULL)
151  return;
152  semicolon[0] = 0; //in colon is now the image type
153  //After the comma the base64 stream starts
154  comma++;
155  int whole_length = strlen(comma);
156  comma[whole_length-1] = 0;
157 
158  bool jpeg = false;
159  if (strcmp(colon, "image/jpeg") == 0)
160  jpeg = true;
161 
162  uint8_t* temp_buffer = video_buffer;
163  if (jpeg)
164  temp_buffer = (uint8_t*)malloc(strlen(comma)+4); //Should always be enough
165 
166  //base64 -> binary data
167  using namespace boost::archive::iterators;
168  typedef
169  transform_width
170  <
171  binary_from_base64
172  <
173  const uint8_t *
174  >,
175  8,
176  6
177  >
178  base64_dec;
179 
180  try
181  {
182  base64_dec src_it(comma);
183  for(int i = 0; i < (whole_length+1)*3/4; ++i)
184  {
185  temp_buffer[i] = *src_it;
186  ++src_it;
187  }
188  }
189  catch (dataflow_exception&)
190  {
191  }
192 
193  if (jpeg)
194  {
195  #if ISAAC_JPEG == 1
196  struct jpeg_decompress_struct cinfo;
197  struct isaac_jpeg_error_mgr jerr;
198  cinfo.err = jpeg_std_error(&jerr.pub);
199  jerr.pub.error_exit = isaac_jpeg_error_exit;
200  if (setjmp(jerr.setjmp_buffer))
201  {
202  jpeg_destroy_decompress(&cinfo);
203  printf("Got invalid jpeg from simulation. Ignoring.\n");
204  free(payload);
205  return;
206  }
207  jpeg_source_mgr src;
208  src.init_source = &isaac_jpeg_init_source;
209  src.fill_input_buffer = &isaac_jpeg_fill_input_buffer;
210  src.skip_input_data = &isaac_jpeg_skip_input_data;
211  src.resync_to_restart = &isaac_jpeg_resync_to_restart;
212  src.term_source = &isaac_jpeg_term_source;
213  jpeg_create_decompress(&cinfo);
214  cinfo.src = &src;
215  cinfo.src->next_input_byte = (JOCTET*)(temp_buffer);
216  cinfo.src->bytes_in_buffer = group->getVideoBufferSize();
217  (void) jpeg_read_header(&cinfo, TRUE);
218  (void) jpeg_start_decompress(&cinfo);
219  int row_stride = cinfo.output_width * cinfo.output_components;
220  JSAMPARRAY buffer = (*cinfo.mem->alloc_sarray)
221  ((j_common_ptr) &cinfo, JPOOL_IMAGE, row_stride, 1);
222  while (cinfo.output_scanline < cinfo.output_height)
223  {
224  int y = cinfo.output_scanline;
225  (void) jpeg_read_scanlines(&cinfo, buffer, 1);
226  for (int x = 0; x < cinfo.output_width; x++)
227  {
228  video_buffer[4*(x+y*cinfo.output_width)+0] = buffer[0][x*3+0];
229  video_buffer[4*(x+y*cinfo.output_width)+1] = buffer[0][x*3+1];
230  video_buffer[4*(x+y*cinfo.output_width)+2] = buffer[0][x*3+2];
231  }
232  }
233  (void) jpeg_finish_decompress(&cinfo);
234  jpeg_destroy_decompress(&cinfo);
235  free(temp_buffer);
236  #else
237  memset( video_buffer, rand()%255, group->video_buffer_size );
238  #endif
239  }
240  free(payload);
241 }

243 std::string Broker::getStream(std::string connector,std::string name,std::string ref)
244 {
245  void* reference;
246  try
247  {
248  reference = (void*)std::stol(ref);
249  }
250  catch (...)
251  {
252  return "";
253  }
254  std::string result ("");
255  ThreadList< InsituConnectorGroup* >::ThreadListContainer_ptr group = insituConnectorGroupList.getFront();
256  while (group)
257  {
258  if (group->t->name == name)
259  {
260  pthread_mutex_lock(&group->t->streams_mutex);
261  std::map< std::string, std::map< void* , std::string > >::iterator it = group->t->streams.find( connector );
262  if (it != group->t->streams.end())
263  {
264  std::map< void* , std::string >::iterator it2 = (*it).second.find( reference );
265  if (it2 != (*it).second.end())
266  result = (*it2).second;
267  }
268  pthread_mutex_unlock(&group->t->streams_mutex);
269  }
270  group = group->next;
271  }
272  return result;
273 }

275 errorCode Broker::run()
276 {
277  printf("Running ISAAC Master\n");
278  signal(SIGINT, sighandler);
279  printf("Starting insitu plugin listener\n");
280  if (insituMaster.init(inner_port))
281  {
282  fprintf(stderr,"Error while starting insitu plugin listener\n");
283  signal(SIGINT, SIG_DFL);
284  return -1;
285  }
286  pthread_create(&insituThread,NULL,Runable::run_runable,&insituMaster);
287 
288  for (auto it = dataConnectorList.begin(); it != dataConnectorList.end(); it++)
289  {
290  printf("Launching %s\n",(*it).connector->getName().c_str());
291  pthread_create(&((*it).thread),NULL,Runable::run_runable,(*it).connector);
292  }
293  for (auto it = imageConnectorList.begin(); it != imageConnectorList.end(); it++)
294  {
295  printf("Launching %s\n",(*it).connector->getName().c_str());
296  pthread_create(&((*it).thread),NULL,Runable::run_runable,(*it).connector);
297  }
298  while (force_exit == 0)
299  {
301  // Iterate over all insitu clients //
303  ThreadList< InsituConnectorContainer* >::ThreadListContainer_ptr insitu = insituMaster.insituConnectorList.getFront();
304  while (insitu)
305  {
306  ThreadList< InsituConnectorContainer* >::ThreadListContainer_ptr next = insitu->next;
307  //Check for new messages for every insituConnector
308  MessageContainer* lastMessage = NULL;
309  if (insitu->t->connector->messagesOut.back)
310  lastMessage = insitu->t->connector->messagesOut.back->t;
311  while (MessageContainer* message = insitu->t->connector->masterGetMessage())
312  {
313  bool delete_message = true;
314  if (message->type == PERIOD)
315  {
316  if (insitu->t->group == NULL) //Later!
317  {
318  delete_message = false;
319  insitu->t->connector->clientSendMessage( message );
320  }
321  else
322  {
323  //Let's see, whether some options are broadcastet and change them in the initData
324  json_t* js;
325  if (json_array_size( js = json_object_get(message->json_root, "projection") ) == 16)
326  json_object_set( insitu->t->group->initData, "projection", js );
327  if (json_array_size( js = json_object_get(message->json_root, "rotation") ) == 9)
328  json_object_set( insitu->t->group->initData, "rotation", js );
329  if (json_array_size( js = json_object_get(message->json_root, "position") ) == 3)
330  json_object_set( insitu->t->group->initData, "position", js );
331  if ( js = json_object_get(message->json_root, "distance") )
332  json_object_set( insitu->t->group->initData, "distance", js );
333  if ( js = json_object_get(message->json_root, "interpolation") )
334  json_object_set( insitu->t->group->initData, "interpolation", js );
335  if ( js = json_object_get(message->json_root, "iso surface") )
336  json_object_set( insitu->t->group->initData, "iso surface", js );
337  if ( js = json_object_get(message->json_root, "step") )
338  json_object_set( insitu->t->group->initData, "step", js );
339  if ( js = json_object_get(message->json_root, "background color") )
340  json_object_set( insitu->t->group->initData, "background color", js );
341  //Filter payload
342  json_t* payload = json_object_get( message->json_root, "payload" );
343  if (payload)
344  {
345  json_incref( payload );
346  json_incref( message->json_root );
347  json_object_del( message->json_root, "payload" );
348  //Allocate, receive and send video
349  uint8_t*video_buffer=(uint8_t*)malloc(insitu->t->group->video_buffer_size);
350  receiveVideo(insitu->t->group,video_buffer,json_dumps( payload , JSON_ENCODE_ANY ));
351  int l=imageConnectorList.size();
352  ImageBufferContainer *container=new ImageBufferContainer(UPDATE_BUFFER,video_buffer,insitu->t->group,l,"",NULL,message->json_root,payload,insitu->t->group->getID());
353  if(l==0)
354  container->image->suicide();
355  else
356  for(auto ic=imageConnectorList.begin();ic!=imageConnectorList.end();ic++)
357  (*ic).connector->masterSendMessage(container);
358  }
359  else
360  {
361  //Direct send of metadata!
362  ThreadList<MetaDataClient*>::ThreadListContainer_ptr dc=dataClientList.getFront();
363  while(dc)
364  {
365  int stream;
366  bool dropable;
367  if(dc->t->doesObserve(insitu->t->group->getID(),stream,dropable))
368  dc->t->masterSendMessage(new MessageContainer(message->type,message->json_root,true,dropable));
369  dc=dc->next;
370  }
371  }
372  }
373  }
374  else
375  if (message->type == REGISTER) //Saving the metadata description for later
376  {
377  //Get group
378  std::string name( json_string_value( json_object_get( message->json_root, "name" ) ) );
379  InsituConnectorGroup* group = NULL;
380  ThreadList< InsituConnectorGroup* >::ThreadListContainer_ptr it = insituConnectorGroupList.getFront();
381  while (it)
382  {
383  if (it->t->name == name)
384  {
385  group = it->t;
386  break;
387  }
388  it = it->next;
389  }
390  if (group == NULL)
391  {
392  group = new InsituConnectorGroup(name);
393  insituConnectorGroupList.push_back( group );
394  }
395  insitu->t->group = group;
396 
397  switch (message->type)
398  {
399  case REGISTER:
400  group->initData = message->json_root;
401  group->framebuffer_width = json_integer_value( json_object_get(group->initData, "framebuffer width") );
402  group->framebuffer_height = json_integer_value( json_object_get(group->initData, "framebuffer height") );
403  group->video_buffer_size = group->framebuffer_width*group->framebuffer_height*4;
404  delete_message = false;
405  printf("New connection, giving id %i (control)\n",insitu->t->connector->getID());
406  break;
407  }
408 
409  group->master = insitu->t;
410  group->id = insitu->t->connector->getID();
411 
412  if (group->master)

415  ThreadList<MetaDataClient*>::ThreadListContainer_ptr dc = dataClientList.getFront();
416  while (dc)
417  {
418  dc->t->masterSendMessage(new MessageContainer(REGISTER,group->initData,true));
419  dc = dc->next;
420  }
421  for (auto it = imageConnectorList.begin(); it != imageConnectorList.end(); it++)
422  (*it).connector->masterSendMessage(new ImageBufferContainer(GROUP_ADDED,NULL,group,1));
423  }
424  }
425  else
426  if (message->type == EXIT_PLUGIN) //Let's tell everybody and remove it from the list
427  {
428  InsituConnectorContainer* insituContainer = insitu->t;
429  int id = insituContainer->connector->getID();
430  printf("Connection %i closed.\n",id);
431  //Group does still exist?
432  if (insituContainer->group)
433  {
434  InsituConnectorGroup* group = insituContainer->group;
435  //Add id of group
436  json_object_set_new( message->json_root, "id", json_integer( group->getID() ) );

438  while (dc)
439  {
440  dc->t->masterSendMessage(new MessageContainer(EXIT_PLUGIN,message->json_root,true));

443  for (auto it = imageConnectorList.begin(); it != imageConnectorList.end(); it++)
444  (*it).connector->masterSendMessage(new ImageBufferContainer(GROUP_FINISHED,NULL,group,1));
445  //Now let's remove the whole group
446  group->master->group = NULL;
447  printf("Removed group %i\n",group->id);

449  while (it)
450  {
451  if (it->t == group)

453  pthread_t thread;
454  pthread_create(&thread,NULL,delete_pointer_later<InsituConnectorGroup>,insituConnectorGroupList.remove(it));
455  break;
456  }
457  it = it->next;
458  }
459  }

461  break;
462  }

466  break;
